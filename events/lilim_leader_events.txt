namespace = lilimleader

## GENERAL LEADER

# Rei Ayanami is set as ruler
country_event = {
	id = lilimleader.1
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {		
		is_country_type = default
		has_civic = civic_divine_rule_rei
	}
	
	immediate = {
		# Happiness in the population
		every_owned_pop = {
			remove_modifier = "pop_ayanami_longing"
		}
		
		leader = {
			set_age = 20
		}
		
		create_species = {
			name = "Lilim"
			class = lilim
			portrait = lilim_admiral
			traits = random
		}						
		
		create_leader = {
			class = admiral
			species = last_created_species
			name = "Rei Ayanami"
			set_age = 20						
			traits = {
				trait = leader_trait_admiral_chosen
			}

			skill = 10
			
		}	
		last_created_leader = {
			set_leader_flag = ayanami_admiral			
		}
	}
}

country_event = {
	id = lilimleader.2
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {		
		is_country_type = default
		has_civic = civic_divine_rule_rei
	}
	
	immediate = {
		if = {
			limit = {
				from = {
					leader_class = admiral
					has_leader_flag = ayanami_admiral
				}
			}
			kill_leader = {
				type = ruler
				show_notification = no
			}
		}
		else_if = {
					limit = {
						from = {
							leader_class = ruler
						}
					}
					random_owned_leader = {			
						limit = {
							leader_class = admiral
							has_leader_flag = ayanami_admiral
						}
						kill_leader = {
							type = admiral
							show_notification = no
					}
				}
			}
		}	
	}
