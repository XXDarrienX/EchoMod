namespace = lion_me_game_start

# cleanup of buildings in home systems with no empire
event = {
	id = lion_me_game_start.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_planet = {
			limit = {
					has_planet_flag = lion_me_home_planet
					is_homeworld = no
			}
			every_tile = {
				limit = { has_blocker = "tb_failing_infrastructure" }
				remove_blocker = yes
				set_blocker = random
			}
			every_tile = {
				limit = { has_blocker = "tb_decrepit_dwellings" }
				remove_blocker = yes
				set_blocker = random
			}
			every_tile = {
				limit = {
					has_building = yes
				}
				remove_building = yes
			}
		}
	}
}
