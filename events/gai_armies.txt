namespace = gai_ai_armies

country_event = {
	id = gai_ai_armies.1
	hide_window = yes
    is_triggered_only = yes 
	
	immediate = {
            if = {
                limit = {
                    is_ai = yes
					count_armies = {
						limit = { army_type = assault_army }
						count < 30
					}
						
					has_monthly_income = {
						resource = energy
						value > 30
					}
					energy > 500
					minerals > 100
                }
                country_event = { id = gai_ai_armies.2 }
            }
        }
	}


country_event = {
	id = gai_ai_armies.2	
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		random_owned_planet = {
			limit = {
				is_occupied_flag = no
				has_orbital_bombardment = no
			}
			while = {
				count = 30
				create_army = {
					owner = ROOT
					type = "assault_army"
				}
			}
		}
	}
}
