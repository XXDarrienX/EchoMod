namespace = gai_ai_armies

event = {
	id = gai_ai_armies.1
	hide_window = yes
    is_triggered_only = yes 
	
	immediate = {
        every_playable_country = {
            if = {
                limit = {
                    is_ai = yes
					is_subject = no
					count_armies = {
						limit = { army_type = assault_army }
						count < 20
					}
					OR = {
						has_monthly_income = {
							resource = energy
							value > -20
						}
						is_at_war = yes
					}
					is_country_type = "default"
					years_passed > 20
                }
                country_event = { id = gai_ai_armies.2 }
            }
        }
	}
}


country_event = {
	id = gai_ai_armies.2	
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		random_owned_planet = {
			limit = {
				is_occupied_flag = no
				has_orbital_bombardment = no
			}
			create_army = {
				owner = ROOT
				type = "assault_army"
			}
		}
	}
}
