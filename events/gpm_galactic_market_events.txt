namespace = gpm_gal_mark

# Enable resources for Galactic Market
event = {
	id = gpm_gal_mark.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_global_flag = galactic_market_founded

		OR = {
			NOT = { is_on_market = gpm_arcane_technology }
		}
	}

	immediate = {
		if = {
			limit = {
									NOT = { is_on_market = gpm_arcane_technology }
				any_playable_country = {
				resource_stockpile_compare = {  resource = gpm_arcane_technology value >= 1 } }
			}
										enable_on_market = gpm_arcane_technology
		}
	}
}
