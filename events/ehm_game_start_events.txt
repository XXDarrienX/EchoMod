namespace = ehm_game_start

### Change homeworld to Gaia World if have Life-Seeded civic on Game Start
country_event = {
	id = ehm_game_start.1
	hide_window = yes
	is_triggered_only = yes

	trigger = { has_civic = civic_hive_world }

	immediate = {
		every_owned_pop = {
			modify_species = {
				species = this
				ideal_planet_class = pc_hive_world
				base = none # makes this replace parent species
			}
		}
		random_owned_planet = {
			limit = { is_capital = yes }
			change_pc = pc_hive_world
			set_planet_size = 25
			reroll_deposits = yes
			reset_start_buildings_and_blockers_life_seeded = yes
		}
		# Nuke other colonies gained from advanced empire, etc
		every_owned_planet = {
			limit = { is_capital = no }
			every_tile = {
				kill_pop = yes
				remove_building = yes
			}
		}
		change_dominant_species = { species = last_created change_all = yes }
	}
}
