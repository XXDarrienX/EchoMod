# Give Hive Minds their special ruler traits at the start if applicable
country_event = {
	id = echo_gestalt_traits.1
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		has_authority = auth_hive_mind
		FROM = { leader_class = ruler }
	}

	immediate = {
        from = {
            if = {
                limit = {
                    has_trait = complex_species_trait_Natural_Flight

                add_ruler_trait = complex_species_trait_Adorable_Species_rul
            }

            else_if = {

            }

            else_if = {

            }

            else_if = {

            else_if = {

            }

            else_if = {

            }

            else_if = {
                limit = {
                    has_trait = trait_additrait_planetary_designers
                }
