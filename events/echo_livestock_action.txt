namespace = echo_livestock

# Add new livestock buildings & remove unused ones
pop_event = {
	id = echo_livestock.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		exists = owner
        owner = {
            OR = {  
            has_policy_flag = slavery_allowed
            has_authority = auth_hive_mind
            }
        }
	}

	immediate = {
		planet = {
			clear_livestock_buildings = yes
			add_livestock_buildings = yes
		}
	}
}
