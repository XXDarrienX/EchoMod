namespace = tfx_ai

# Assigns modifier to AI 
 
 country_event = {
	id = tfx_ai.101
	hide_window = yes
 	
	is_triggered_only = yes
	
	trigger = { 
		owner = { is_ai = yes }
	}		
	
	immediate = {
		log = "---- AI ARTIFACT EVENT FIRED tfx_ai.101 ----"
		random_list = {
			#Artifact 1: The Black Orb
			25 = { 
				modifier = {
					factor = 0
					has_global_flag = tf_global_black_orb
				}
				set_global_flag = tf_global_black_orb
				set_country_flag =  tf_ai_orb
				add_modifier = {
					modifier = tf_black_orb_modifier 
					days = -1
				}
			}
			#Artifact 2: The Endless Shard
			25 = { 
				modifier = {
					factor = 0
					has_global_flag = tf_global_endless_shard
				}
				set_global_flag = tf_global_endless_shard
				set_country_flag = tf_ai_shard
				add_modifier = {
					modifier = tf_endless_shard_modifier 
					days = -1
				}
			}
			# Artifact 3: Soul Crystal
			25 = { 
				modifier = {
					factor = 0
					has_global_flag = tf_global_soul_crystal
				}

				set_global_flag = tf_global_soul_crystal
				set_country_flag = tf_ai_crystal
				add_modifier = {
					modifier = tf_soul_crystal_modifier
					days = -1
				}
			}
			# Nothing is recovered
			25 = { set_country_flag = tf_ai_nothing }
		}
	}
	
	after = {
		switch = {
			trigger = has_country_flag
			tf_ai_nothing = { log = "---- AI Empty Handed ----" } 
			tf_ai_orb = { log = "---- AI Black Orb ----" }
			tf_ai_shard = { log = "---- AI Endless Shard ----" }
			tf_ai_crystal = { log = "---- AI Soul Crystal ----" }
			default = { log = "---- AI Nothing Happened ----"}
		}
	}
 }
