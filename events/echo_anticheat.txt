namespace = echo_anticheat

## Kills Country for using a restricted trait
event = {
    id = echo_anticheat.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        is_multiplayer = yes
    }

    immediate = {
        every_playable_country = {
            limit = {
                is_ai = no
            }
            every_owned_pop_species = {
                limit = {
                    OR = {
                        has_trait = trait_creative
                        has_trait = trait_fragile
                        has_trait = trait_quarian
                        has_trait = trait_serpentoid
                        has_trait = trait_geth_synthetic
                        has_trait = trait_krogan
                        has_trait = trait_asari_lifespan
                        has_trait = trait_asari_selection
                    }
                }
            }
            destroy_country = yes
        }
    }
}
