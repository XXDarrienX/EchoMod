namespace = utopia

# Handle leader traits after species modification
country_event = {
	id = utopia.2508
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		every_owned_leader = {
			add_leader_traits_after_modification = yes
            remove_leader_traits_after_modification = yes
			echo_remove_traits = yes
			echo_add_traits = yes
		}
		
		every_pool_leader = {
			add_leader_traits_after_modification = yes
            remove_leader_traits_after_modification = yes
			echo_remove_traits = yes
			echo_add_traits = yes			
		}
	}		
}
