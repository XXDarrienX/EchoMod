# Investment Return - Part 0
# On start - Start the Investment
country_event = {
	id = wiirlak_ac.0010
	hide_window = yes
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		has_civic = wiirlak_ac_civic_investment
	}
	
	immediate = {
		country_event = { id = wiirlak_ac.0011 days = 3600 }
	}
}

# Investment Return - Part 1
# 10 years Mark - Give the return on Investment
country_event = {
	id = wiirlak_ac.0011
	title = wiirlak_ac_investment1_event_title
	desc = wiirlak_ac_investment1_event_desc
	picture = GFX_evt_board_meeting
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		has_civic = wiirlak_ac_civic_investment
	}
	
	option = {
		name = wiirlak_ac_event_b
		add_resource = {
			energy = 500
			minerals = 250
			alloys = 100
			influence = 100
		}
	}
	option = {
		name = wiirlak_ac_event_c
		add_resource = {
			energy = 1500
			minerals = 750
			alloys = 250
			influence = 250
		}
		hidden_effect = { country_event = { id = wiirlak_ac.0012 days = 9000 } }
	}
}

# Shrouded Connection Origin
# On start - Give the planet modifier & change the moon into a Shrouded type

		}
	}
}

# Nyctophobia Origin
# On start - Add the Penal Colonies tech & modifies the original moon into a jungle penal colony
country_event = {
	id = wiirlak_ac.3020
	hide_window = yes
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		has_origin = wiirlak_ac_origin_nyctophobia
	}
	
	immediate = {
		give_technology = { message = no tech = tech_penal_colonies }
		capital_scope = {
			random_moon = {
				change_pc = pc_tropical
				add_deposit = d_tropical_island
				add_deposit = d_lush_jungle
				add_deposit = d_fertile_lands
				
				add_deposit = d_geothermal_vent
				add_deposit = d_rushing_waterfalls

				add_deposit = d_mineral_fields
				add_deposit = d_submerged_ore_veins

				add_deposit = d_dense_jungle
				add_deposit = d_dense_jungle
				add_deposit = d_dense_jungle
				add_deposit = d_toxic_kelp
				add_deposit = d_noxious_swamp

				create_colony = { owner = ROOT }
				create_pop = { species = ROOT }
				add_modifier = { modifier = penal_colony }
				remove_deposit = d_energy_5
			}
		}
	}
}
