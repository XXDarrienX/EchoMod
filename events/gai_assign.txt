namespace = gai_assign

event = {
	id = gai_assign.1
	hide_window = yes
    is_triggered_only = yes 
	
	immediate = {
        every_playable_country = {
			IF = {
				limit = {
					is_ai = yes
					is_country_type = default
				}
				IF = {
					limit = {
						any_owned_fleet = {
							is_ship_class = shipclass_science_ship
							NOT = { exists = leader }
						}
					}
					random_owned_leader = {
						limit = {
							leader_class = scientist
							NOR = {
								is_researching_area = society
								is_researching_area = engineering
								is_researching_area = physics
								has_leader_flag = busy_sci
							}
						}
						save_event_target_as = assign_me
						set_leader_flag = busy_sci
					}
					random_owned_fleet = {
						limit = {
							is_ship_class = shipclass_science_ship
							NOT = { exists = leader }
						}
						assign_leader = event_target:assign_me
					}
				}
			}
		}
	}
}
