namespace = ehm_buildings

# Replace some buildings for Hive-Minded empires
# this: capital planet
# from: founder species
planet_event = {
	id = ehm_buildings.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = { limit = { owner = { has_authority = auth_hive_mind } }
			# Hive capital
			every_tile = {
				limit = { has_building = building_capital_1 }
				remove_building = yes
				set_building = building_hive_mind_capital_1
			}
		}
	}
}
