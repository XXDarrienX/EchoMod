namespace = precursor

# Vultaum Home System Discovered (HIDDEN)
country_event = {
	id = precursor.98
	hide_window = yes
	
	fire_only_once = yes
	is_triggered_only = yes
	
	immediate = {
		set_global_flag = vultaum_system_discovered
		set_country_flag = vultaum_system
		if = {
			limit = {
				any_system_within_border = { has_star_flag = empire_home_system }
			}
			random_system_within_border = {
				limit = { has_star_flag = empire_home_system }
				spawn_system = {
					min_distance = 10
					max_distance = 30
					max_jumps = 0
					initializer = "vultaumar_system"
				}
			}
		}
		else = {
			random_system = {
				limit = { has_star_flag = empire_home_system }
				spawn_system = {
					min_distance = 10
					max_distance = 30
					max_jumps = 0
					initializer = "vultaumar_system"
				}
			}
		}
		add_extra_hyperlane_to_spawned_system_effect = yes
		country_event = { id = precursor.100 days = 2 }
	}
}
