namespace = raiding

@raiding_gain_min = 50
@raiding_gain_low = 100
@raiding_gain_med = 250
@raiding_gain_max = 500

planet_event = {
	id = raiding.1
	
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_orbital_bombardment = yes
		has_orbital_bombardment_stance = raiding
	}
	
	immediate = {		
		if = {
			limit = {
				OR = {
					has_building = building_colony_shelter
				}
			}
			root.From = { 
				add_food = @raiding_gain_min
				add_minerals = @raiding_gain_min
				add_energy = @raiding_gain_min
			}
		}
		if = {
			limit = {
				OR = {
					has_building = building_capital_1
				}
			}
			root.From = { 
				add_food = @raiding_gain_low 
				add_minerals = @raiding_gain_low 
				add_energy = @raiding_gain_low 
			}
		}
		if = {
			limit = {
				OR = {
					has_building = building_capital_2
					has_building = building_hab_capital
				}
			}
			root.From = { 
				add_food = @raiding_gain_med
				add_minerals = @raiding_gain_med
				add_energy = @raiding_gain_med
			}	
		}			
		if = {
			limit = {
				has_building = building_capital_3
			}
			root.From = { 
				add_food = @raiding_gain_max 
				add_minerals = @raiding_gain_max
				add_energy = @raiding_gain_max
			}
		}	
	}
}
