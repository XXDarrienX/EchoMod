namespace = eve_empire_modifiers

event = {
    id = eve_empire_modifiers.1
    hide_window = yes
    is_triggered_only = yes
   
    immediate = {
        every_country = {
            limit = { has_country_flag = amarr }
				change_government = {
					civics = {
						civic = civic_amarr
						civic = civic_slaver_guilds
					}
				}
						add_modifier = {
						modifier = amarr_modifier
						days = -1
						}
					}

				}
			}
