wg_crusade = {
	casus_belli = cb_crusade
	#GFX = "GFX_diplomacy_crusade"
	
	hide_if_no_cb = no
	release_occupied_systems_on_status_quo = yes
	
	surrender_acceptance = -100
	
	war_exhaustion = 0.75

	potential = {
		from = { is_country_type = default }
	}
	possible = {
		always = yes
	}
	on_status_quo = {

			from = {
				add_threat = {
					who = root
					amount = 0.10
				}				
			}			
		}
	on_accept = {
		custom_tooltip = wg_crusade_accept
		custom_tooltip = wg_crusade_accept_2
		if = {
			limit = {
				from = { has_federation = yes }
			}
			custom_tooltip = wg_crusade_accept_3
		}
		hidden_effect = {		
			from = { copy_ethos_and_authority = root }
			end_rivalry = from
			from = { end_rivalry = root }
			every_system_within_border = {
				limit = {
					from = { has_claim = prev }
				}
				remove_claims = { who = from }
			}			
			every_country = {
				limit = { 	
					OR = {
						is_in_federation_with = root 
						AND = {
							is_subject = yes
							exists = overlord
							overlord = { is_same_value = root }
						}
					}
				}
				end_rivalry = from
				from = { end_rivalry = prev }	
				every_system_within_border = {
					limit = {
						from = { has_claim = prev }
					}
					remove_claims = { who = from }
				}
			}
			from = {
				add_threat = {
					who = root
					amount = 0.10
				}				
			}
			from = {
				leave_alliance = { override_requirements = yes }
			}
		}
	}
	
	ai_weight = {
		weight = 5
	}
}
