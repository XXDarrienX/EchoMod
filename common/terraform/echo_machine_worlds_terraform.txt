	to = "pc_machine"
	from = "pc_nuked"

	duration = @machineTime

		OR = { 
			has_origin = origin_machine
			has_ascension_perk = ap_machine_worlds

				}
			}
		}
		modifier = {

				is_colony = no
			}
		}
	}
}
