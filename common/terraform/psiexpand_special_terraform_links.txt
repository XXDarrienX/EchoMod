	duration = 7200

	potential = {
		from = { has_modifier = terraforming_candidate }
	}

	effect = {
		from = {
			add_modifier = {
				modifier = terraforming_candidate
				days = -1
			}
		}
	}
	
	ai_weight = {
		weight = 0
	}
}
