	}

			trigger = {
				is_organic_species = yes

		}

		produces = {
			trigger = {

			society_research = 2
			unity = 1
		}

		produces = {
			trigger = {
				exists = owner
				owner = { has_valid_civic = civic_vb_psi_soul_eater }
			}
			soul_crystals = 2
		}

		produces = {
			trigger = {
				exists = owner
				is_robotic_species=  no
				owner = { has_valid_civic = civic_vb_psi_soul_eater }
				OR = {
					has_trait = trait_latent_psionic
					has_trait = trait_psionic
				}
			}
			soul_crystals = 4
		}

		produces = {
			trigger = {
				is_energy_being = yes
			}
			energy = 4
		}
		produces = {
			trigger = {
				is_silicon = yes
			}
			minerals = 4
		}
	}
		
	weight = {
		weight = @special_job_weight
	}
}
