### Each edict has a regular effect, and a stronger/cheaper one depending on the civics of the Hive.
# Research
country_edict = {
	name = "overgrow_research_drones"
	length = @EdictDuration
	cost = { influence = @Edict3Cost }

	modifier = {
		all_technology_research_speed = 0.1
	}

	potential = {
		has_authority = auth_hive_mind
		NOT = { has_valid_civic = civic_hive_natural_neural_network }

		weight = 0
	}
}
