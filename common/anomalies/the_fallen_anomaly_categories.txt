# Probe Encountered 
# > This anomaly will have a chance to spawn a number of different events 

TF_PROBE_ENCOUNTERED_CAT = {
	desc = TF_PROBE_ENCOUNTERED_CAT_DESC
	picture = GFX_evt_satellite_in_orbit
	level = 3 
	
	spawn_chance = {
		modifier ={
			add = 3
			is_planet_class = pc_barren
		}
	}
	
	on_success = { 
		1 = tf_anomaly_probe.100
		2 = { 
			anomaly_event = tf_anomaly_probe.200
			max_once_global = yes
		}
	}
}

#Creates Legendary Weapon Anomaly

TF_LEGENDARY_WEAPON_CAT = {
	desc = TF_LEGENDARY_WEAPON_CAT_DESC
	picture = GFX_evt_medieval_alien_civilization
	level = 4
	
	max_once = yes

	spawn_chance = { 
		modifier = {
			add = 300
			is_ideal_planet_class = { 
				who = FROM.owner
				status = yes
			}
		}
	}

	spawn_chance = { 
		modifier = {
			factor = 0
			FROM.owner = {
				OR = { 
					has_authority = auth_machine_intelligence
					has_authority = auth_hive_mind
				}
			}
		}
	}

	on_success = {
		1 = tf_anomaly_01.1
	}
}
