gai_build_fleet = {
	OR = {
		AND = {
			is_at_war = yes
			used_naval_capacity_percent < 0.95
		}
		AND = {
			is_at_war = no
			has_monthly_income = {
				resource = energy
				value > 10
			}
			has_monthly_income = {
				resource = alloys
				value > 15
			}
			used_naval_capacity_percent < 0.95
		}
		AND = {
			is_at_war = no
			used_naval_capacity_percent < 0.95
			OR = {
				used_naval_capacity_percent < 0.50
				highest_threat > 45
			}
		}
	}
}

gai_build_district ={
	planet = {
		has_building_construction = no
		NOT = { has_modifier = living_sea }
		OR = {
			free_jobs < 2
			any_owned_pop = { 
				NOT = { has_citizenship_type = { type = citizenship_organic_trophy } }
				OR = { 
					is_unemployed = yes 
					has_job = servant
				}
				NOR = {
					is_pop_category = specialist
					is_pop_category = ruler
				}
			}
		}
	}
}

gai_build_power_district ={
	planet = {
		has_building_construction = no
		NOT = { has_modifier = living_sea }
		OR = {
			free_jobs < 2
			any_owned_pop = { 
				NOT = { has_citizenship_type = { type = citizenship_organic_trophy } }
				OR = { 
					is_unemployed = yes 
					has_job = servant
				}
				NOR = {
					is_pop_category = specialist
					is_pop_category = ruler
					is_pop_category = robot_servant
				}
			}
		}
	}
}

gai_stop_pop = {
	OR = {
		owner = { is_ai = no }
		AND = {
			owner = { is_ai = yes }
			free_housing < 1
			free_district_slots = 0
			free_building_slots = 0
			count_pops >= 75
		}
	}
}

gai_start_pop = {
	OR = {
		owner = { is_ai = no }
		AND = {
			owner = {
				is_ai = yes
			}
			free_housing > 0
			count_pops < 75
		}
	}
}

gai_build_building = {
	OR = {
		planet = {
			OR = { 
				AND = {
					free_district_slots < 3
					num_uncleared_blockers = 0
					owner = { is_ai = yes }
				}
				is_capital = yes
			}
		}
		owner = { is_ai = no }
	}
}


gai_eligible_worker_job = {
	OR = {
		AND = {
			is_unemployed = no
			is_pop_category = specialist
		}
		AND = {	
			is_unemployed = no
			is_pop_category = ruler
		}
	}
}

gai_eligible_specalist_job = {
	OR = {
		is_unemployed = yes
		AND = {
			is_unemployed = no
			OR = { 
				is_pop_category = specialist
				is_pop_category = robot_servant
				is_pop_category = slave
			}
		}
	}
}

gai_eligible_ruler_job = {
	OR = {
		is_unemployed = yes
		AND = {
			is_unemployed = no
			OR = { 
				is_pop_category = ruler
				AND = {
					owner = { has_technology = tech_synthetic_workers }		
					owner = { has_policy_flag = ai_full_rights }
				}
			}
		}
	}
}


gai_special_job = {
	OR = {
		owner = { is_ai = no }
		AND = {
			owner = { is_ai = yes }
			OR = {
				is_pop_category = complex_drone
				AND = {	
					is_unemployed = yes
					is_pop_category = simple_drone
				}
			}
		}
	}
}
