plentiful_factions_mystics = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_spiritualist
	
	unique = yes
	support_multiplier = 1
	
	resources = {
		category = pop_factions
		produces = {
			influence = 2
		}		
	}

	is_potential = {
		has_tradition = tr_plentiful_mysticism_adopt
		NOT = { is_gestalt = yes }
	}
	
	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}
	
	can_join_faction = {
		is_enslaved = no
		is_shackled_robot = no
		is_being_purged = no
		has_culture_shock = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_spiritualist
			has_ethic = ethic_fanatic_spiritualist
		}
		is_robot_pop = no
	}
	
	attraction = {
		base = 150
		
		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}
		
		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
	}
	
	leader = {
		base = 100
		modifier = {
			factor = 0
			species = {
				has_trait = "trait_mechanical"		
			}			
		}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}		
	}

	demand = { # "MYSTICS_BIOAFFLUENCE"
		title = "MYSTICS_BIOAFFLUENCE"
		unfulfilled_title = "TRADITIONALIST_ROBOTS"
		desc = "MYSTICS_BIOAFFLUENCE_DESC"
		
		unfulfilled_effect = -10
		fulfilled_effect = 5
		
		potential = {
			exists = owner
		}
		
		trigger = {
			owner = {
				has_policy_flag = robots_outlawed
			}
		}
	}
	
	demand = { # "MYSTICS_PILGRIMAGE"
		title = "MYSTICS_PILGRIMAGE"
		unfulfilled_title = "MYSTICS_PILGRIMAGE_NO"
		desc = "MYSTICS_PILGRIMAGE_DESC"

		unfulfilled_effect = 0
		fulfilled_effect = 5

		potential = {
			exists = owner
			owner = {
				OR = {
					has_tradition = tr_plentiful_mysticism_1
					has_tradition = tr_plentiful_mysticism_2
					has_tradition = tr_plentiful_mysticism_3
					has_tradition = tr_plentiful_mysticism_4
					has_tradition = tr_plentiful_mysticism_5
				}
			}
		}

		trigger = {
			owner = {
			any_owned_planet = {
					OR = {
						has_building = building_generosity_sanctum
						has_building = building_vigor_sanctum
						has_building = building_frugality_sanctum
						has_building = building_vision_sanctum
						has_building = building_plentiful_traditions_sanctum_of_keys
					}
				}
			}
		}
	}	

	demand = { # "MYSTICS_REDEMPTION"
		title = "MYSTICS_REDEMPTION"
		unfulfilled_title = "MYSTICS_REDEMPTION_NO"
		desc = "MYSTICS_REDEMPTION_DESC"

		unfulfilled_effect = 0
		fulfilled_effect = 5

		potential = {
			exists = owner
			owner = {
				has_surveyed_class = pc_nuked
			}
		}

		trigger = {
			owner = {
				any_owned_planet = {
					is_planet_class = pc_nuked
				}
			}
		}
	}

	demand = { # "MYSTICS_CITADEL"
		title = "MYSTICS_CITADEL"
		unfulfilled_title = "MYSTICS_CITADEL_NO"
		desc = "MYSTICS_CITADEL_DESC"
		
		fulfilled_effect = 5
		unfulfilled_effect = 0
		
		potential = {
			exists = owner
			owner = {
				has_technology = tech_transcendent_faith
			}
		}
		
		trigger = {
			owner = {
			any_owned_planet = {
				OR = {
					has_building = building_citadel_of_faith
					has_building = building_psychic_remembrance_institute
					}
				}
			}
		}
	}

	demand = { # "MYSTICS_PROFIT"
		title = "MYSTICS_PROFIT"
		unfulfilled_title = "MYSTICS_PROFIT_GREED"
		desc = "MYSTICS_PROFIT_DESC"

		unfulfilled_effect = -5
		fulfilled_effect = 5

		potential = {
			exists = owner
			owner = {
				has_technology = tech_galactic_markets
			}
		}

		trigger = {
			owner = {
				any_owned_planet = {
					has_building = building_galactic_stock_exchange
				}
			}
		}
	}

	demand = { # "MYSTICS_VITALITY"
		title = "MYSTICS_VITALITY"
		unfulfilled_title = "MYSTICS_VITALITY_NO"
		desc = "MYSTICS_VITALITY_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = 0

		potential = {
			exists = owner
			owner = {
				is_researching_technology = tech_nano_vitality_crops
			}
		}

		trigger = {
			owner = {
				has_technology = tech_nano_vitality_crops
			}
		}
	}

	demand = { # "MYSTICS_STOCKPILE"
		title = "MYSTICS_STOCKPILE"
		unfulfilled_title = "MYSTICS_STOCKPILE_NO"
		desc = "MYSTICS_STOCKPILE_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -5

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				has_country_resource = {
					type = food
					amount >= 5000
				}
			}
		}
	}

	demand = { # "MYSTICS_SYNTHS"
		title = "MYSTICS_SYNTHS"
		unfulfilled_title = "MYSTICS_SYNTHS_NO"
		desc = "MYSTICS_SYNTHS_DESC"

		unfulfilled_effect = -5
		fulfilled_effect = 10

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
			any_neighbor_country = {
					is_gestalt = no
					is_country_type = default
					any_owned_pop_species = {
						NOT = { has_trait = trait_mechanical }
					}
				}
			}
		}
	}
	
	on_create = {
		pop_faction_event = { id = factions.1 }
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_spiritualist
			}
			remove_modifier = promoted_spiritualist
		}
		if = {
			limit = {
				has_modifier = suppressed_spiritualist
			}
			remove_modifier = suppressed_spiritualist
		}
	}
	
	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"
			
			cost = {
				influence = 500
			}
			
			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}
			
			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_spiritualist" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_spiritualist
					hidden_effect = {
						add_modifier = { modifier = embraced_faction_timer days = 3600 }
						every_pop_faction = {
							limit = { NOT = { is_same_value = root } }
							add_modifier = { modifier = embraced_another_faction days = 3600 }
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default			
								has_communications = prev
							}
							country_event = { id = factions.2000 }
						}
					}				
				}				
			}
			
			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50
					owner = {
						has_ethic = ethic_spiritualist
					}
				}
				modifier = {
					factor = 0
					owner = { has_valid_civic = civic_fanatic_purifiers }
				}				
				modifier = {
					factor = 0
					NOT = { 
						owner = {
							exists = ruler
							ruler = {
								leader_of_faction = plentiful_factions_mystics
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_factions = {
							count < 4
						}					
					}
				}				
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_spiritualist
						has_modifier = promoted_spiritualist
					}
				}
			}

			effect = {
				parameter:empire = {
					add_modifier = { modifier = promoted_spiritualist days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_spiritualist
				}
			}

			effect = {
				parameter:empire = {
					remove_modifier = promoted_spiritualist
				}
			}
			
			ai_weight = {
				base = 0
			}
		}		
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_spiritualist
						has_modifier = promoted_spiritualist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_spiritualist days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_spiritualist
				}
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_spiritualist
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
	}
}
