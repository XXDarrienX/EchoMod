# Arcology Project
decision_arcology_project = {
	owned_planets_only = yes
	icon = decision_arcology_project

	enactment_time = 3600
	resources = {
		category = decisions
		cost = {
			influence = 200
			minerals = 20000
		}
	}

	potential = {
		NOR = {
			is_planet_class = pc_city
			is_planet_class = pc_relic
			has_modifier = resort_colony
			has_modifier = penal_colony
			has_modifier = slave_colony
			has_modifier = pm_wenkwort_custodian
		}
		habitable_structure = no
		exists = owner
		owner = { has_ascension_perk = ap_arcology_project }
		owner = { NOT = {has_civic = civic_urban_idyll} }
	}

	allow = {
		custom_tooltip = {
			fail_text = decision_arcology_project_requirements
			free_district_slots < 1
			num_districts = { type = district_farming value < 1 }
			num_districts = { type = district_mining value < 1 }
			num_districts = { type = district_generator value < 1 }
		}
		num_uncleared_blockers < 1
	}

	effect = {
		# Make city-planet
		custom_tooltip = decision_arcology_project_effects
		hidden_effect = {
			clear_deposits = yes
			remove_modifier = industrialized_planet
			change_pc = pc_city
			set_planet_entity = {
				entity = "city_planet_entity"
				graphical_culture = owner
			}
			owner = {
				country_event = { id = distar.201 } # port for living planet distar.197, triggers in event
			}
			planet_event = { id = mega.200 }
			if = {
				limit = {
					OR = {
						is_active_resolution = "resolution_ecology_environmental_control_board"
						is_active_resolution = "resolution_ecology_paradise_initiative"
					}
				}
				owner = {
					set_timed_country_flag = { flag = resolution_breached_terraformed days = 3600 }
				}
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

decision_industrialize = {
	icon = decision_industrialize
	sound = event_administrative_work
	owned_planets_only = yes
	enactment_time = 480
	resources = {
			category = decisions
			cost = {
				minerals = 800
			}
		}
	potential = {
		habitable_structure = no
		NOR = { 
			is_planet_class = pc_relic
			is_planet_class = pc_city
			is_planet_class = pc_hive
			is_planet_class = pc_machine
			has_modifier = "industrialized_planet"
			has_modifier = "industrialized_arcology"
		}
		exists = owner
		owner = { 
			AND = {
				has_civic = civic_manufactory
				is_regular_empire = yes 
			}
		}
		
	}
	allow = {
		NOR = {
			has_modifier = "industrialized_planet"
			has_modifier = "industrialized_arcology"
		}
	}
	effect ={
		custom_tooltip = decision_industrialize_effects
		
		hidden_effect = {
			
				add_deposit = d_manufactory
				add_modifier = { modifier = "industrialized_planet" days = -1 }
				random_list = {
				1 = {
					# sometimes construction doesn't always go as planned
					add_planet_devastation = 20
					add_deposit = d_radioactive_wasteland
				}
				4 = {
					add_deposit = d_failing_infrastructure
					add_deposit = d_failing_infrastructure
				}
				2 = {
					modifier = {
						factor = 0
						OR = {
							num_districts = { type = district_farming value = 0 }
							num_districts = { type = district_farming value > 5 }
						}
					}
					remove_district = district_farming
					add_deposit = d_failing_infrastructure
				}
				2 = {
					modifier = {
						factor = 0
						num_districts = { type = district_farming value < 5 }
					}
					remove_district = district_farming
					remove_district = district_farming
				}
		
				}
			
			
				
		}			
	}
	
	ai_weight = {
		weight = 0
	}
	
}
