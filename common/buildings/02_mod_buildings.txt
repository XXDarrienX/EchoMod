building_xeno_corps = {
	base_buildtime = 480
	base_cap_amount = 1	

	category = government
	
	
	potential = {
		exists = planet
		planet = { is_capital = yes }
		owner = { has_ascension_perk = ap_arcane_tech }
		NOT = { has_modifier = slave_colony }
	}
	
	destroy_trigger = {
		exists = owner
		NOT = {
			owner = { has_ascension_perk = ap_arcane_tech }
		}
		has_modifier = slave_colony
	}
	
	allow = {
		has_major_upgraded_capital = yes
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_xeno_arcane_add = 2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_xeno_drone_add = 2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {	
			job_xeno_bot_add = 2
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 2
		}
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_xeno_arcane_effect_desc
	}	
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_hive_empire = yes } 
		}
		text = job_xeno_drone_effect_desc
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_machine_empire = yes } 
		}
		text = job_xeno_bot_effect_desc
	}
}

building_alloy_enrichment_plant = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = resource

	destroy_trigger = {
		is_resort_planet = yes
	}

	potential = {
		is_resort_planet = no
		OR = {
			has_building = building_foundry_1
			has_building = building_foundry_2
			has_building = building_foundry_3
		}
	}

	allow = {
		buildings_simple_allow = yes
	}

	planet_modifier = {
		planet_jobs_alloys_produces_mult = 0.15
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		cost = {
			trigger = {
				owner = { has_authority = auth_machine_intelligence }
			}
			minerals = 100
			alloys = 100
		}
		upkeep = {
			energy = 2
		}
	}

	prerequisites = {
		tech_alloy_enrichment_1
	}

	ai_resource_production = {
		alloys = 1
		trigger = {
			always = yes
		}
	}
}
