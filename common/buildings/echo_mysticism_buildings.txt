	base_buildtime = 180
    base_cap_amount = 1
    category = unity

    potential = {
        exists = owner
		owner = {
			has_tradition = tr_plentiful_mysticism_1
		}
    }

    destroy_trigger = {
        exists = owner
		owner = {
			NOT = { has_tradition = tr_plentiful_mysticism_1 }
		}
    }

    resources = {
        category = planet_buildings
        cost = {
            energy = 600
        }
        upkeep = {
            energy = 4
        }
    }
		
	triggered_planet_modifier  = {
        potential = {
            exists = owner
            owner = {
                is_regular_empire = yes
                has_tradition = tr_plentiful_mysticism_1
            }
        }
        planet_jobs_unity_produces_mult = 0.2
        job_technician_add = 2
    }
    
    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_gestalt = yes
                has_tradition = tr_plentiful_mysticism_1
            }
        }
        planet_jobs_unity_produces_mult = 0.2
        job_technician_drone_add = 2
    }

    triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_technician_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_technician_effect_desc
	}
	
	ai_weight = {
		weight = 1
	}
}
