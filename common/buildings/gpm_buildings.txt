gpm_precursor_research_center = {
	base_buildtime = 480
	base_cap_amount = 1
	category = research
	
	potential = {
		exists = owner
		NOR = {
			is_resort_planet = yes
			has_modifier = slave_colony
		}
		has_ancrel = yes		
	}

	allow = {
		is_capital = yes
	}

	destroy_trigger = {
		exists = planet
		planet = {
			is_capital = no
		}
	}

	prerequisites = {
		gpm_tech_arcane_research
	}


	resources = {
		category = planet_buildings
		cost = {
			minerals = 700
		}
		upkeep = {
			energy = 5
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_gpm_precursor_archaeologist_add = 2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner			
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_gpm_precursor_archaeologist_drone_add = 2
		}
	}
	
	triggered_desc = {
		text = job_gpm_precursor_archaeologist_effect_desc
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_gpm_precursor_archaeologist_drone_effect_desc
	}
	
	on_queued = {
		owner = {
			subtract_variable = { which = gpm_precursor_research_center_build_count value = 1 }
		}
	}

	on_built = {
		owner = { 
			set_country_flag = gpm_precursor_research_center_built
			remove_country_flag = gpm_precursor_research_center_queued
		}
	}

	on_destroy = {
		owner = {
			remove_country_flag = gpm_precursor_research_center_built
		}
	}
}
