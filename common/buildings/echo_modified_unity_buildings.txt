# Spiritualists
building_temple = {
	base_buildtime = 240
	base_cap_amount = 1

	category = unity
	
	potential = {
		owner = { is_spiritualist = yes }
		NOT = { has_modifier = slave_colony }
	}
	
	planet_modifier = {
		job_priest_add = 2
		pop_ethic_spiritualist_attraction_mult = 0.05
    }
    
    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                has_civic = civic_monastery_culture
            }
        }
        modifier = {
            job_miner_add = 1
            job_technician_add = 1
            job_farmer_add = 1
            pop_housing_usage_mult = -0.05
        }
        
    }
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		upkeep = {
			energy = 1
		}		
	}
	
	upgrades = {
		building_holotemple
	}
	
	prerequisites = {
		"tech_cultural_heritage"
	}
	
	show_tech_unlock_if = {
		is_spiritualist = yes
	}
	
	triggered_desc = {
		text = job_priest_effect_desc
    }
    
    triggered_desc = {
		exists = owner
		owner = {
			has_civic = civic_monastery_culture
		}
        text = job_farmer_effect_desc
    }

    triggered_desc = {
		exists = owner
		owner = {
			has_civic = civic_monastery_culture
		}
        text = job_miner_effect_desc
    }

	ai_weight = {
		weight = 1
	}
}
