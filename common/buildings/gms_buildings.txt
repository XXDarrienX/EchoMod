

building_small_administration_ai = {
	base_buildtime = 360
	empire_unique = yes
	
	cost = {
		minerals = 2500
		energy = 250
		influence = 25
	}
	
	potential = {
		planet = {
			OR = {
				is_planet_class = pc_habitat
				is_planet_class = pc_ringworld_habitable
			}
		}
	}
	
	destroy_if = {
		OR = {
			owner = { has_policy_flag = ai_outlawed }
			planet = { is_capital = yes}
		}
	}
	
	allow = {
		custom_tooltip = {
			fail_text = "requirement_not_homeworld"
			planet = {
				is_capital = no
			}
		}
		custom_tooltip = {
			fail_text = "needs_right_policies"
			owner = { NOT = {has_policy_flag = ai_outlawed }}
		}
	}
	
	required_resources = {
		energy = 5
	}
	
	prerequisites = {
		"ig_tech_small_administration_ai"
	}
	
	ai_weight = {
		weight = 40
	}
	
	produced_resources = {
		unity = 6
	}
	
	country_modifier = {
		all_technology_research_speed = 0.05
	}
	
	planet_modifier = {
		planet_sensor_range_add = 1
		planet_building_cost_mult = -0.2
		planet_building_upkeep_mult = -0.2
		planet_army_build_cost_mult = -0.25
		planet_army_build_speed_mult = 0.6
	}
}
