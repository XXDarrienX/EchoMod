# Queen's Den - Special Authority building for Hive Mind
building_queen_den = {
	base_buildtime = 180
	is_listed = yes
	empire_unique = yes
	cost = {
		minerals = 100
	}

	required_resources = { 
        energy = 3 
	}
		
    planet_modifier = { 
        tile_resource_food_mult = 0.2
        tile_resource_unity_mult = 0.2
	}
	
	produced_resource_trigger = {
		modifier = {
			has_civic = civic_hive_devouring_swarm
			resources = {
				unity = 25
			}
		}
	}
	
	produced_resources = { 
         unity = 10
    }
	
	potential = { owner = { has_authority = auth_hive_mind } }
	
	destroy_if = {
		exists = owner
		owner = { NOT = { has_authority = auth_hive_mind } }
    }
    
    ai_weight = {weight = 100 }
}
