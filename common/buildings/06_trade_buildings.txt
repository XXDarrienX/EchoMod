building_commercial_zone = {
	base_buildtime = 360

	category = trade
	
	potential = {
		NOT = { is_planet_class = pc_habitat }
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}
	
	destroy_trigger = {
		owner = { is_regular_empire = no }
	}
	
	planet_modifier = {
		job_clerk_add = 5
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}	
		upkeep = {
			energy = 2
		}		
	}
	
	prerequisites = {
		"tech_interplanetary_commerce"
	}
	
	show_tech_unlock_if = { 
		is_regular_empire = yes
	}
	
	triggered_desc = {
		text = job_clerk_effect_desc
	}
	
	upgrades = {
		"building_commercial_megaplex"
	}
	
	ai_weight = {
		weight = 2
		modifier = 
		{
			factor = 5
			owner = { 
				has_monthly_income = {
					resource = energy
					value < 10
				}
			}
			free_amenities < 2
		}
		modifier = {
			factor = 0
			free_jobs > 2
#			count_owned_pops = { 
#				limit = { 
#					OR = { 
#						is_unemployed = yes 
#						has_job = servant 
#					}
#				}
#				count < 3
#			}
		}
		modifier = {
			factor = 0
			OR = {
				num_buildings = { type = building_commercial_zone value >= 2 }
				num_buildings = { type = building_commercial_megaplex value >= 0 }
			}
		}
	}

	ai_resource_production = {
		energy = 1
		trigger = {
			always = yes
		}
	}
}
