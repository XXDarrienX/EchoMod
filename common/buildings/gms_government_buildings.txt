building_gms_govseat = {
	base_buildtime = 180
	icon = building_gms_govseat
	base_cap_amount = 1
	category = government

	potential = {
		exists = owner
		owner = {
			is_gestalt = no
			NOR = { 
				has_country_flag = building_gms_govseat_queued
				has_country_flag = building_gms_govseat_built
			}
		}
		has_upgraded_capital = yes
	}

	allow = {
		exists = owner
		owner = {
			NOT = { has_country_flag = building_gms_govseat_queued }
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			is_gestalt = yes
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 5000
			influence = 200
		}
	}

	planet_modifier = {
		job_government_official_add = 1
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_xenophobe = yes }		
		}
		text = govseat_xenophobe_effect_desc
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_xenophile = yes }		
		}
		text = govseat_xenophile_effect_desc
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_militarist = yes }		
		}
		text = govseat_militarist_effect_desc
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_pacifist = yes }		
		}
		text = govseat_pacifist_effect_desc
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_spiritualist = yes }		
		}
		text = govseat_spiritualist_effect_desc
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_materialist = yes }		
		}
		text = govseat_materialist_effect_desc
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_authoritarian = yes }		
		}
		text = govseat_authoritarian_effect_desc
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_egalitarian = yes }		
		}
		text = govseat_egalitarian_effect_desc
	}

	show_tech_unlock_if = { is_gestalt = no }

	on_queued = { owner = { set_country_flag = building_gms_govseat_queued } }
	on_unqueued = { owner = { remove_country_flag = building_gms_govseat_queued } }
	on_built = { 
		owner = { 
			remove_country_flag = building_gms_govseat_queued
			set_country_flag = building_gms_govseat_built
		}
	}
	on_destroy = { owner = { remove_country_flag = building_gms_govseat_built } }
		
	prerequisites = {
		"ig_tech_government"
	}

	ai_weight = {
		weight = 100
	}
}
