	empire_unique = yes
	
	cost = {
		minerals = 5000
		influence = 100
	}

		planet = { sector_controlled = no }
		owner = { is_xenophobe = yes }
	}

	show_tech_unlock_if = { is_xenophobe = yes }

		custom_tooltip = {

	}

		country_starbase_influence_cost_mult = -0.2
		country_claim_influence_cost_mult = -0.1
	}
	
	prerequisites = {
		"ig_tech_government"
	}
	
	upgrades = {

	}
}
