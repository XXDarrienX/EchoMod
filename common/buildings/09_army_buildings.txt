building_military_academy = {
	base_buildtime = 480
	base_cap_amount = 1

	category = army
	
	potential = {
		always = yes
	}
	
	allow = {
		has_upgraded_capital = yes
	}
	
	army_modifier = {
		army_starting_experience_add = 100
	}
	
	triggered_planet_modifier = {
		potential = {
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}			
		upkeep = {
			energy = 2
		}
	}
	
	prerequisites = {
		"tech_centralized_command"
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
	
	ai_weight = {
		weight = 0.5
		modifier = {
			factor = 0
			free_jobs > 2
			NOT = { owner = { has_valid_civic = civic_machine_servitor }}
#			count_owned_pops = { 
#				limit = { 
#					OR = { 
#						is_unemployed = yes 
#						has_job = servant 
#					}
#				}
#				count < 1
#			}
		}
		modifier = {
			factor = 0
			owner = { has_valid_civic = civic_machine_servitor }
			OR = {
				AND = {
					NOT = { any_pop = { has_living_standard = { type = living_standard_organic_trophy }}}
					free_jobs > 2
				}
				AND = {
					any_pop = { has_living_standard = { type = living_standard_organic_trophy }}
					OR = {
						free_jobs > 4
						any_pop = {
							is_species_class = MACHINE
							is_unemployed = yes
						}
					}
				}
			}
		}
	}
}
