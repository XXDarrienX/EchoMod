building_food_processing_center = {
can_be_ruined = no
	base_buildtime = 480
	can_build = no

	category = resource
		
	planet_modifier = {
		planet_farmers_food_produces_mult = 0.25
	}
	
	allow = {
		has_upgraded_capital = yes
	}
	
	potential = {
		gai_has_motes = yes
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_farmer_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_agri_drone_add = 2
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
			volatile_motes = 50
		}
		upkeep = {
			energy = 1
			volatile_motes = 1
		}
	}	
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_agri_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_farmer_effect_desc
	}
	
	prerequisites = {
		"tech_food_processing_2"
	}
	
	upgrades = {
		building_psychic_hydroponics_complex
	}
	
	ai_weight = {
		weight = 2
		modifier = {
			factor = 5
			owner = { 
				has_monthly_income = {
					resource = food
					value < 5
				}
			}
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}
