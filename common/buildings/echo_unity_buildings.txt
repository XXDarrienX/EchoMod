building_temple = {
	base_buildtime = @b1_time

	category = unity

	potential = {
		exists = owner
		building_basic_income_check = yes
		owner = {
			is_spiritualist = yes
			NOR = {
				has_valid_civic = civic_death_cult
				has_valid_civic = civic_death_cult_corporate
			}
		}
		NOT = { has_modifier = slave_colony }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				OR = {
					has_valid_civic = civic_death_cult
					has_valid_civic = civic_death_cult_corporate
					is_spiritualist = no
				}
			}
			has_modifier = slave_colony
		}
	}

	convert_to = {
		building_hive_node
		building_simulation_1
		building_autochthon_monument
		building_corporate_monument
		building_sacrificial_temple_1
	}

	planet_modifier = {
		job_priest_add = @b1_jobs
		pop_ethic_spiritualist_attraction_mult = 0.05
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_civic = civic_monastery_culture }	
		}
		planet_housing_add = 1
		job_priest_add = 1
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	upgrades = {
		building_holotemple
	}

	prerequisites = {
		"tech_cultural_heritage"
	}

	show_tech_unlock_if = {
		is_spiritualist = yes
		NOT = {
			OR = {
				has_valid_civic = civic_death_cult
				has_valid_civic = civic_death_cult_corporate
			}
		}
	}

	triggered_desc = {
		text = job_priest_effect_desc
	}

	ai_weight = {
		weight = 25
		modifier = {
			factor = 0
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value < 8
				}
			}
		}
		modifier = {
			factor = 0.1
			OR = {
				has_building = building_temple
				has_building = building_holotemple
				has_building = building_sacred_nexus
			}
		}
	}
}

# Servitors
building_organic_sanctuary = {
	base_buildtime = @b1_time

	category = unity

	potential = {
		exists = owner
		owner = { has_valid_civic = civic_machine_servitor }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				NOT = { owner = { has_valid_civic = civic_machine_servitor } }
			}
			AND = {
				owner = { is_ai = yes }
				count_pops = { limit = { has_citizenship_type = { type = citizenship_organic_trophy } } count <= 20 }
				num_buildings = { type = building_organic_sanctuary value > 3 }
			}
			AND = {
				owner = { is_ai = yes }
				count_pops = { limit = { has_citizenship_type = { type = citizenship_organic_trophy } } count <= 40 }
				num_buildings = { type = building_organic_paradise value > 5 }
			}
		}
	}

	convert_to = {
		building_luxury_residence
		building_communal_housing
	}

	planet_modifier = {
		planet_housing_add = 10
		job_bio_trophy_add = 10
		job_maintenance_drone_add = 1
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	upgrades = {
		"building_organic_paradise"
	}

	ai_weight = {
		weight = 0.5
		modifier = {
			factor = 0
			NOT = { owner = { has_valid_civic = civic_machine_servitor } }
		}
		modifier = {
			weight = 1000
			any_owned_pop = {
				is_unemployed = yes
				has_citizenship_type = { type = citizenship_organic_trophy }
			}
		}
		modifier = {
			weight = 250
			OR = {
				has_building = building_factory_1
				has_building = building_factory_2
				has_building = building_factory_3
			}
			NOR = {
				has_building = building_organic_sanctuary
				has_building = building_organic_paradise
			}
			planet = {
				owner = {
					any_owned_pop = {
						is_organic_species = yes
						prevprev = { habitability = { who = prev value > 0.5 } }
					}
				}
			}
		}
	}
}
