building_hydroponics_farm = {
can_be_ruined = no
	base_buildtime = 360

	category = resource
	
	potential = {
		planet = {
			or = {
				free_building_slots > 0
				has_building = building_hydroponics_farm
				owner = { is_ai = no }
			}
		}
		NOT = { is_planet_class = pc_machine }
		NOT = { has_modifier = resort_colony }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_farmer_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_agri_drone_add = 2
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			energy = 2
		}
	}	
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_agri_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_farmer_effect_desc
	}
	
	prerequisites = {
		"tech_hydroponics"
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 50
			gai_good_food = yes
		}
		modifier = {
			weight = 1500
			gai_low_food = yes
			planet = { free_district_slots = 0 }
		}
		modifier = {
			weight = 5000
			owner = { 
				has_monthly_income = {
					resource = food
					value < 100
				}
			}
			planet = { is_planet_class = pc_habitat }
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}
