building_stronghold = {
can_be_ruined = no
	base_buildtime = 240
	add_to_first_building_slot = yes

	category = army
	
	potential = {
		always = yes
	}
	
	planet_modifier = {
		planet_housing_add = 1
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = 1
		}
	}

	triggered_planet_modifier = {
        potential = {
			exists = owner
            owner = {
                has_tradition = tr_plentiful_defence_5
            }
        }
        modifier = {
            planet_housing_add = 5
        }
    }

	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}			
		upkeep = {
			energy = 1
		}
	}
	
	prerequisites = {
		"tech_planetary_defenses"
	}
	
	upgrades = {
		"building_fortress"
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}	
	
	ai_weight = {
		weight = 0.5
		modifier = {
			factor = 0
			count_owned_pops = { 
				limit = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant 
					}
				}
				count < 1
			}
		}
	}
}
