building_slave_processing = {
can_be_ruined = no
	base_buildtime = 360
	base_cap_amount = 1

	category = government

	potential = {
		planet = {
			or = {
				free_building_slots > 0
				has_building = building_slave_processing
				owner = { is_ai = no }
			}
		}
		owner = { is_regular_empire = yes }
		owner = { has_policy_flag = slavery_allowed }
		NOT = { has_modifier = resort_colony }
	}
	
	allow = {
		has_upgraded_capital = yes
	}
	
	planet_modifier = {
		planet_jobs_slave_produces_mult = 0.10
		pop_cat_slave_political_power = -0.25
	}	

	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}
		upkeep = {
			energy = 2		
		}
	}
	
	prerequisites = {
		"tech_neural_implants"
	}

	upgrades = {
		building_plentiful_traditions_advanced_slave_processing
	}
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 10
			any_owned_pop = { is_enslaved = yes }
		}
	}
}

	}

	}

		is_regular_empire = yes
	}
	
	upgrades = {
		building_hall_judgment
	}
	
	ai_weight = {
		weight = 1

			factor = 10

		}

		modifier = {

		}

		modifier = {

		}
	}
