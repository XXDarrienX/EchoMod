building_holotemple = {
can_be_ruined = no
	base_buildtime = 360
	can_build = no

	category = unity
	
	potential = {
		owner = { is_spiritualist = yes }
		NOT = { has_modifier = slave_colony }
	}
	
	convert_to = {
		building_network_junction
		building_hive_cluster
		building_heritage_site
		building_corporate_site
	}

	destroy_trigger = {
		owner = { is_spiritualist = no }
	}

	
	allow = {
		has_upgraded_capital = yes
	}
	
	planet_modifier = {
		job_priest_add = 4
		pop_ethic_spiritualist_attraction_mult = 0.10
	}

    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                has_civic = civic_monastery_culture
            }
        }
        job_miner_add = 2
        job_farmer_add = 2
        pop_housing_usage_mult = -0.1
    }
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}
		upkeep = {
			energy = 2
		}		
	}
	
	upgrades = {
		building_sacred_nexus
	}
	
	prerequisites = {
		"tech_holographic_rituals"
	}
	
	triggered_desc = {
		text = job_priest_effect_desc
	}

	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { has_civic = civic_monastery_culture }		
		}
        text = job_farmer_effect_desc
    }

    triggered_desc = {
		trigger = { 
			exists = owner
			owner = { has_civic = civic_monastery_culture }		
		}
        text = job_miner_effect_desc
    }

	ai_weight = {
		weight = 1
		# No one to work it
		modifier = {
			factor = 0
			count_owned_pops = { 
				limit = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant 
					}
				}
				count < 2
			}
		}
	}
}
