building_foundry_1 = {
	base_buildtime = 360

	category = manufacturing
	
	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_alloy_drone_add = 2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_fabricator_add = 2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_foundry_add = 2
		}
	}

	triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = { 
                is_gestalt = no
                has_tradition = tr_plentiful_industry_4
            }
        }
        modifier = {
            job_foundry_add = 1
        }
    }

    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = { 
                is_hive_empire = yes
                has_tradition = tr_plentiful_industry_4
            }
        }
        modifier = {
            job_alloy_drone_add = 1
        }
    }

    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = { 
                is_machine_empire = yes
                has_tradition = tr_plentiful_industry_4
            }
        }
        modifier = {
            job_fabricator_add = 1
        }
    }
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}
		upkeep = {
			energy = 4
		}
	}
	
	upgrades = {
		building_foundry_2
	}

	prerequisites = {
		tech_basic_industry
	}	
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_regular_empire = yes } 
		}
		text = job_foundry_effect_desc
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_hive_empire = yes } 
		}
		text = job_alloy_drone_effect_desc
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_machine_empire = yes } 
		}
		text = job_fabricator_effect_desc
	}
	
	ai_weight = {
		weight = 1
		modifier = 
		{
			factor = 6
			owner = { 
				OR = {
					AND = {
						is_machine_empire = no
						has_monthly_income = {
							resource = alloys
							value < 15
						}
					}
					AND = {
						is_machine_empire = yes
						has_monthly_income = {
							resource = alloys
							value < 22
						}
					}
				}
			}
		}
		modifier = {
			factor = 0
			OR = {
				AND = {
					is_capital  = no
					OR = {
						num_buildings = { type = building_foundry_1 value >= 2 }
						num_buildings = { type = building_foundry_2 value >= 1 }
						num_buildings = { type = building_foundry_3 value >= 1 }
					}
				}
				AND = {
					is_capital  = yes
					OR = {
						num_buildings = { type = building_foundry_1 value >= 3 }
						num_buildings = { type = building_foundry_2 value >= 1 }
						num_buildings = { type = building_foundry_3 value >= 1 }
					}
				}
			}
		}
		modifier = {
			factor = 0
			free_jobs > 2
			NOT = { owner = { has_valid_civic = civic_machine_servitor }}
#			count_owned_pops = { 
#				limit = { 
#					OR = { 
#						is_unemployed = yes 
#						has_job = servant 
#					}
#				}
#				count < 1
#			}
		}
		modifier = {
			factor = 0
			owner = {
				has_valid_civic = civic_machine_servitor
			}
			OR = {
				AND = {
					NOT = { any_pop = { has_living_standard = { type = living_standard_organic_trophy }}}
					free_jobs > 2
				}
				AND = {
					any_pop = { has_living_standard = { type = living_standard_organic_trophy }}
					OR = {
						free_jobs > 4
						any_pop = {
							is_species_class = MACHINE
							is_unemployed = yes
						}
					}
				}
			}
		}
		
		modifier = 
		{
			factor = 0
			owner = { 
				OR = {
					AND = {
						is_machine_empire = no
						has_monthly_income = {
							resource = minerals
							value < 24
						}
					}
					AND = {
						is_machine_empire = yes
						has_monthly_income = {
							resource = minerals
							value < 32
						}
					}
				}
			}
		}
	}

	ai_resource_production = {
		alloys = 1
		trigger = {
			always = yes
		}
	}
}
