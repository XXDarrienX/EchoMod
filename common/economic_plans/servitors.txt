# Early ---------------------------------------

early_servitor_plan = {
	type = early

	income = {
		energy = 50
		minerals = 60
		food = 25
		consumer_goods = 25 
		alloys = 50
		unity = 50

		physics_research = 200
		society_research = 200
		engineering_research = 200
		
		exotic_gases = 3
		volatile_motes = 3
		rare_crystals = 3

	}

	focus = {
		energy = 30
		minerals = 50
		alloys = 25
		food = 10
		consumer_goods = 10
		unity = 50
		
		exotic_gases = 1
		volatile_motes = 1
		rare_crystals = 1
	}

	pops = 500
	empire_size = 2

	potential = {
		years_passed > 7
		years_passed < 40
		has_valid_civic = civic_machine_servitor
		OR = {
			has_monthly_income = {
				resource = energy
				value < 2
			}
			has_monthly_income = {
				resource = minerals
				value < 50
			}
			NOR = {
				any_owned_planet = {
					is_under_colonization = no
					is_controlled_by = root
					free_amenities < 4
					free_jobs < 3
					count_pops = {
						limit = {
							complex_specialist_job_check_trigger = yes
						}
						count > 12
					}
				}
				any_owned_pop = {
					is_unemployed = yes
					has_citizenship_type = { type = citizenship_organic_trophy }
				}
				any_owned_planet = {
					free_building_slots > 0
					NOR = { 
						has_building = building_organic_sanctuary 
						has_building = building_organic_paradise
					}
				}
			}
		}
	}

	ai_weight = {
		weight = 1
	}
}
