		is_planet_class = pc_hive
		num_districts = { type = district_hive value < 4 }
		free_jobs < 1
	}

	prio_districts = {
		district_hive
	}

	buildings = {
		1 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}

		2 = {
			building = building_spawning_pool
			available = {
				always = yes
			}
		}

		3 = {
			building = building_clone_vats
			available = {
				always = yes
			}
		}

		4 = {
			building = building_mineral_purification_plant
			available = {
				always = yes
			}
		}

		5 = {
			building = building_sentinel_posts
			available = {
				num_buildings = { type = building_sentinel_posts value < 2 }
			}
		}

		6 = {
			building = building_refinery
			available = {
				OR = {
					AND = {
						owner = { NOT = { has_ascension_perk = ap_engineered_evolution } }
						num_buildings = { type = building_refinery value < 6 }
					}

					AND = {
						owner = { has_ascension_perk = ap_engineered_evolution }
						num_buildings = { type = building_refinery value < 5 }
					}
				}
			}
		}

		7 = {
			building = building_chemical_plant
			available = {
				num_buildings = { type = building_chemical_plant value < 5 }
			}
		}

		8 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}
	}
}
