ssg_hive_hiveworld_farming_planet = {
	available = {
		has_designation = col_farming
		owner = { is_hive_empire = yes }
		is_planet_class = pc_hive
		num_districts = { type = district_hive value < 5 }
		free_jobs < 1
	}

	prio_districts = {
		district_hive
	}

	buildings = {
		1 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}

		2 = {
			building = building_spawning_pool
			available = {
				always = yes
			}
		}

		3 = {
			building = building_clone_vats
			available = {
				always = yes
			}
		}

		4 = {
			building = building_sentinel_posts
			available = {
				num_buildings = { type = building_sentinel_posts value < 2 }
			}
		}

		5 = {
			building = building_food_processing_facility
			available = {
				always = yes
			}
		}

		6 = {
			building = building_hydroponics_farm
			available = {
				OR = {
					AND = {
						num_buildings = { type = building_hydroponics_farm value < 11 }
						owner = { NOT = { has_ascension_perk = ap_engineered_evolution } }
					}

					AND = {
						num_buildings = { type = building_hydroponics_farm value < 10 }
						owner = { has_ascension_perk = ap_engineered_evolution }
					}
				}
			}
		}

		7 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}
	}
}
