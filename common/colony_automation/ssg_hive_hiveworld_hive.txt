ssg_hive_hiveworld_hive_planet = {
	available = {
		has_designation = col_hive
		owner = { is_hive_empire = yes }
		is_planet_class = pc_hive
		num_districts = { type = district_hive value < 10 }
		free_jobs < 1
	}

	prio_districts = {
		district_hive
	}

	buildings = {
		1 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}

		2 = {
			building = building_spawning_pool
			available = {
				always = yes
			}
		}

		3 = {
			building = building_sentinel_posts
			available = {
				OR = {
					num_buildings = { type = any value = 2 }
					num_buildings = { type = any value = 3 }
					num_buildings = { type = any value = 4 }
					num_buildings = { type = any value = 5 }
				}
			}
		}

		4 = {
			building = building_mineral_purification_plant
			available = {
				num_districts = { type = district_mining_uncapped value > 6 }
			}
		}

		5 = {
			building = building_research_lab_1
			available = {
				num_buildings = { type = any value > 5 }
			}
		}

		6 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}
	}
}
