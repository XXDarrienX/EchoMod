
ssg_hive_habitat_fortress_planet = {
	available = {
		has_designation = col_habitat_fortress
		owner = { has_authority = auth_hive_mind }
		NOR = {
			has_planet_flag = research_habitat
			has_planet_flag = mining_habitat
		}
		free_jobs < 1
	}

	prio_districts = {
		district_hab_energy
	}

	buildings = {
		1 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}

		2 = {
			building = building_spawning_pool
			available = {
				always = yes
			}
		}

		3 = {
			building = building_clone_vats
			available = {
				always = yes
			}
		}

		4 = {
			building = building_sentinel_posts
			available = {
				num_buildings = { type = building_sentinel_posts value < 2 }
			}
		}

		5 = {
			building = building_planetary_shield_generator
			available = {
				always = yes
			}
		}

		6 = {
			building = building_energy_grid
			available = {
				always = yes
			}
		}

		7 = {
			building = building_stronghold
			available = {
				OR = {
					owner = { NOT = { has_ascension_perk = ap_engineered_evolution } }

					AND = {
						has_building = building_clone_vats
						owner = { has_ascension_perk = ap_engineered_evolution }
					}

					AND = {
						owner = { NOT = { has_ascension_perk = ap_engineered_evolution } }
						NOT = { has_building = building_clone_vats }
						num_buildings = { type = any value < 15 }
					}
				}
			}
		}

		8 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}
	}
}
