ssg_hive_ringworld_generator_planet = {
	available = {
		has_designation = col_ring_generator
		owner = { is_hive_empire = yes }
		num_districts = { type = district_rw_hive value < 1 }
		free_jobs < 1
	}

	prio_districts = {
		district_rw_hive
	}

	buildings = {
		1 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}

		2 = {
			building = building_spawning_pool
			available = {
				always = yes
			}
		}

		3 = {
			building = building_clone_vats
			available = {
				always = yes
			}
		}

		4 = {
			building = building_energy_grid
			available = {
				always = yes
			}
		}

		5 = {
			building = building_sentinel_posts
			available = {
				num_buildings = { type = any value < 5 }
			}
		}

		6 = {
			building = building_research_lab_1
			available = {
				OR = {
					AND = {
						owner = { has_ascension_perk = ap_engineered_evolution }
						num_buildings = { type = any value > 8 }
					}

					AND = {
						owner = { NOT = { has_ascension_perk = ap_engineered_evolution } }
						num_buildings = { type = any value > 7 }
					}
				}
			}
		}

		7 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}
	}
}
