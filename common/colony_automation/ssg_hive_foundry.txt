							AND = {
								num_buildings = { type = any value > 9 }
								owner = { NOT = { has_ascension_perk = ap_engineered_evolution } }

						planet_size = 13

							AND = {
								num_buildings = { type = any value > 13 }
								owner = { NOT = { has_ascension_perk = ap_engineered_evolution } }

							AND = {
								num_buildings = { type = any value > 14 }
								owner = { has_ascension_perk = ap_engineered_evolution }
							}
						}	
					}

					AND = {
						planet_size = 14
						num_buildings = { type = any value > 14 }
						owner = { NOT = { has_ascension_perk = ap_engineered_evolution } }

				}
			}

			building = building_hive_capital
			available = {
				always = yes
			}
