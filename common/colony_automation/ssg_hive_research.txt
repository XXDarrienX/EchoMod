ssg_hive_research_planet = {
	available = {
		has_designation = col_research
		owner = { is_hive_empire = yes }
		NOT = { is_planet_class = pc_hive }
		planet_size < 16
		free_jobs < 1
	}

	prio_districts = {
		district_hive
	}

	buildings = {
		1 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}

		2 = {
			building = building_spawning_pool
			available = {
				always = yes
			}
		}

		3 = {
			building = building_clone_vats
			available = {
				always = yes
			}
		}

		4 = {
			building = building_sentinel_posts
			available = {
				OR = {
					AND = {
						planet_size < 12
						num_buildings = { type = building_sentinel_posts value < 2 }
					}

					AND = {
						planet_size > 11
						planet_size < 16
						num_buildings = { type = building_sentinel_posts value < 3 }
					}
				}
			}
		}

		5 = {
			building = building_research_lab_1
			available = {
				OR = {
					AND = {
						planet_size < 12
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 3 }
								num_buildings = { type = any value < 11 }
							}

							AND = {
								has_building = building_clone_vats
								num_buildings = { type = any value > 4 }
								num_buildings = { type = any value < 12 }
							}
						}
					}

					AND = {
						planet_size = 12
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 4 }
								num_buildings = { type = any value < 13 }
							}

							AND = {
								has_building = building_clone_vats
								num_buildings = { type = any value > 5 }
								num_buildings = { type = any value < 14 }
							}
						}
					}

					AND = {
						planet_size = 13
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 4 }
								num_buildings = { type = any value < 14 }
							}

							AND = {
								has_building = building_clone_vats
								num_buildings = { type = any value > 5 }
								num_buildings = { type = any value < 15 }
							}
						}
					}

					AND = {
						planet_size = 14
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 4 }
								num_buildings = { type = any value < 15 }
							}

							AND = {
								has_building = building_clone_vats
								num_buildings = { type = any value > 5 }
								num_buildings = { type = any value < 16 }
							}
						}
					}

					AND = {
						planet_size = 15
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 4 }
								num_buildings = { type = any value < 15 }
							}

							AND = {
								has_building = building_clone_vats
								num_buildings = { type = any value > 5 }
								num_buildings = { type = any value < 16 }
							}
						}
					}
				}
			}
		}

		6 = {
			building = building_refinery
			available = {
				OR = {
					AND = {
						planet_size < 12
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 10 }
							}

							AND = {
								has_building = building_clone_vats
								num_buildings = { type = any value > 11 }
							}
						}
					}

					AND = {
						planet_size = 12
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 12 }
							}

							AND = {
								has_building = building_clone_vats
								num_buildings = { type = any value > 13 }
							}
						}
					}

					AND = {
						planet_size = 13
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 13 }
							}

							AND = {
								has_building = building_clone_vats
								num_buildings = { type = any value > 14 }
							}
						}
					}

					AND = {
						planet_size = 14
						OR = {
							AND = {
								NOT = { has_building = building_clone_vats }
								num_buildings = { type = any value > 14 }
							}
						}
					}
				}
			}
		}

		7 = {
			building = building_hive_capital
			available = {
				always = yes
			}
		}
	}
}
