ssg_machine_mining_planet = {
	available = {
		has_designation = col_mining
		owner = { is_machine_empire = yes }
		NOT = { is_planet_class = pc_machine }
		OR = {
			num_districts = { type = district_nexus value < 8 }

			AND = {
				num_districts = { type = district_nexus value = 8 }
				num_free_districts = { type = district_generator value = 0 }
				num_free_districts = { type = district_mining value = 0 }
				num_free_districts = { type = district_farming value = 0 }
			}
		}
		free_jobs < 1
	}

	prio_districts = {
		district_nexus
	}

	buildings = {
		1 = {
			building = building_deployment_post
			available = {
				always = yes
			}
		}

		2 = {
			building = building_machine_assembly_plant
			available = {
				always = yes
			}
		}

		3 = {
			building = building_sentinel_posts
			available = {
				OR = {
					AND = {
						planet_size < 16
						num_buildings = { type = building_sentinel_posts value < 1 }
					}

					AND = {
						planet_size < 22
						num_buildings = { type = building_sentinel_posts value < 2 }
					}

					AND = {
						planet_size > 21
						num_buildings = { type = building_sentinel_posts value < 2 }
					}
				}
			}
		}

		4 = {
			building = building_mineral_purification_plant
			available = {
				always = yes
			}
		} 

		5 = {
			building = building_research_lab_1
			available = {
				OR = {
					AND = {
						planet_size < 16
						num_buildings = { type = any value > 3 }
						num_buildings = { type = any value < 7 }
					}

					AND = {
						planet_size < 22
						num_buildings = { type = any value > 4 }
						num_buildings = { type = any value < 9 }
					}

					AND = {
						planet_size > 21
						num_buildings = { type = any value > 4 }
						num_buildings = { type = any value < 10 }
					}
				}
			}
		}

		6 = {
			building = building_crystal_plant
			available = {
				OR = {
					AND = {
						planet_size < 16
						num_buildings = { type = any value > 6 }
						num_buildings = { type = building_crystal_plant value < 1 }
					}

					AND = {
						planet_size < 22
						num_buildings = { type = any value > 8 }
						num_buildings = { type = building_crystal_plant value < 1 }
					}

					AND = {
						planet_size > 21
						num_buildings = { type = any value > 9 }
						num_buildings = { type = building_crystal_plant value < 1 }
					}
				}
			}
		}

		7 = {
			building = building_refinery
			available = {
				OR = {
					AND = {
						planet_size < 16
						num_buildings = { type = any value > 6 }
						num_buildings = { type = building_refinery value < 5 }
					}

					AND = {
						planet_size < 22
						num_buildings = { type = any value > 8 }
						num_buildings = { type = building_refinery value < 5 }
					}

					AND = {
						planet_size > 21
						num_buildings = { type = any value > 9 }
						num_buildings = { type = building_refinery value < 5 }
					}
				}
			}
		}

		8 = {
			building = building_chemical_plant
			available = {
				OR = {
					AND = {
						planet_size < 16
						num_buildings = { type = any value > 6 }
						num_buildings = { type = building_chemical_plant value < 3 }
					}

					AND = {
						planet_size < 22
						num_buildings = { type = any value > 8 }
						num_buildings = { type = building_chemical_plant value < 1 }
					}
				}
			}
		}

		9 = {
			building = building_deployment_post
			available = {
				always = yes
			}
		}
	}
}
