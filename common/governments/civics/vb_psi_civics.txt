#Soul Eater
civic_vb_psi_soul_eater = {
	potential = { ethics = { NOT = { value = ethic_gestalt_consciousness } } }
	playable = { host_has_dlc = "Utopia" }
	modification = no
	icon = "gfx/interface/icons/governments/civics/vb_psi_soul_eaters_easy_black.dds"
	possible = {
		ethics = {
			OR = {
				text = civic_tooltip_xenophobe
				value = ethic_xenophobe
				value = ethic_fanatic_xenophobe
			}
			OR = {
				text = civic_tooltip_spiritualist
				value = ethic_spiritualist
				value = ethic_fanatic_spiritualist
			}
			NOR = {
				text = civic_tooltip_not_pacifist
				value = ethic_pacifist
				value = ethic_fanatic_pacifist
			}
        }
        authority = {
            NOT = {
                value = auth_corporate
            }
        }
		civics = {
			NOR = {
				value = civic_fanatic_purifiers
				value = civic_vb_psi_touched_by_the_shroud
			}
		}
		origin = {
			NOT = {
				value = origin_syncretic_evolution
			}
		}
	}
	
	random_weight = {
		base = 30
		modifier = {
			factor = 0
			NOT = { host_has_dlc = "Utopia" }
		}
		modifier = {
			# civic has no effect after galaxy generation, disable for random generation
			factor = 0
			has_global_flag = game_started
		}
	}
	
	description = "civic_vb_psi_soul_eater_effects"
	modifier = {
		leader_skill_levels = 5
		slave_market_cost_mult = -0.10
		planet_jobs_slave_produces_mult = 0.10
		country_base_soul_crystals_produces_add = 10
	}
	
	traits = {
		trait = trait_vb_psi_soul_eater
	}
}
