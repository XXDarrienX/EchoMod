minerals_expenditure_armies = {
	resource = minerals 
	type = expenditure
	category = armies
	
	potential = {
		count_armies = { limit = { NOR = { 
			army_type = defense_army
			army_type = robotic_defense_army
			army_type = machine_defense
			}}
			count < 20 
		}
		any_neighbor_country = {
			is_rival = prev
			is_hostile_to = prev
		}
		has_monthly_income = {
			resource = minerals
			value > 30
		}
		has_resource = { type = minerals amount > 999 }
		OR = {
			NOT = { any_owned_pop = { is_unemployed = yes } }
			is_at_war = yes
		}
	}
	
	fraction = {
		weight = 0.00
	}	
	static_min = {
		base = 10000
	}
}
