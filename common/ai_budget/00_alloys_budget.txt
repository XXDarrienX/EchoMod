alloys_expenditure_buffer = { # Buffer, will not be spent
	resource = alloys 
	type = expenditure
	category = buffer
	
	potential = {
		always = no
	}
	
	fraction = {
		weight = 0.0
	}	
}

alloys_expenditure_ships = {
	resource = alloys 
	type = expenditure
	category = ships
	
	potential = {
		gai_build_fleet = yes
		
		OR = {
			AND = {
				has_resource = { type = alloys amount > 325 }
				OR = {
					years_passed > 9
					used_naval_capacity_integer < 3
				}
			}
			is_at_war = yes
		}
	}
	
	fraction = {
		weight = 0.00
	}
	
	static_min = {
		base = 10000
	}

	static_max = {
		base = 10000
	}
}
