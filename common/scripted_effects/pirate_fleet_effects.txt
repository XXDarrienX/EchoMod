create_normal_pirate_country = {
	event_target:pirate_system = {
		random_system_planet = {
			save_event_target_as = pirate_home
			create_country = {
				name = "NAME_Pirates"
				type = pirate
				species = event_target:owner_species
				name_list = "PRT1"
				flag = {
					background = {
						category = "backgrounds"
						file = "00_solid.dds"
					}
					colors ={
						"black"
						"black"
						"null"
						"null"
					}
				}
				effect = {
					save_event_target_as = pirate_band
				}
			}
			event_target:pirate_band = {
				set_name = random
				randomize_flag_symbol = "pirate"
				every_country = {
					establish_communications_no_message = prev
				}
			}
		}
	}
}

create_pirate_fleet = {
	event_target:pirate_band = {
	if = {limit = { is_difficulty < 2 }	
		if = { limit = { NOT = { has_modifier = basic_debuff_1 } } add_modifier = { modifier = basic_debuff_1 days = -1 } }	
    }	
	if = {limit = { is_difficulty > 2 }	
		if = { limit = { years_passed > 24 NOT = { has_modifier = basic_buff_1 } } add_modifier = { modifier = basic_buff_1 days = -1 } }
		if = { limit = { years_passed > 49 NOT = { has_modifier = basic_buff_3 } } add_modifier = { modifier = basic_buff_3 days = -1 } }
		if = { limit = { years_passed > 74 NOT = { has_modifier = basic_buff_5 } } add_modifier = { modifier = basic_buff_5 days = -1 } }
        if = { limit = { years_passed > 109 NOT = { has_modifier = basic_buff_7 } } add_modifier = { modifier = basic_buff_7 days = -1 } }	
        if = { limit = { years_passed > 154 NOT = { has_modifier = basic_buff_9 } } add_modifier = { modifier = basic_buff_9 days = -1 } }
	}
	if = {limit = { is_difficulty > 3 } 
		if = { limit = { years_passed > 34 NOT = { has_modifier = basic_buff_2 } } add_modifier = { modifier = basic_buff_2 days = -1 } }	
		if = { limit = { years_passed > 59 NOT = { has_modifier = basic_buff_4 } } add_modifier = { modifier = basic_buff_4 days = -1 } }		
        if = { limit = { years_passed > 84 NOT = { has_modifier = basic_buff_6 } } add_modifier = { modifier = basic_buff_6 days = -1 } }			
        if = { limit = { years_passed > 119 NOT = { has_modifier = basic_buff_8 } } add_modifier = { modifier = basic_buff_8 days = -1 } }		
        if = { limit = { years_passed > 184 NOT = { has_modifier = basic_buff_10 } } add_modifier = { modifier = basic_buff_10 days = -1 } }	
    }		
    if = { limit = { years_passed < 40 }	
		create_leader = {
			class = admiral
			species = owner_main_species
			name = random
			skill = 1
			traits = {
				trait = random_trait
			}
		}
	}
    else_if = { limit = { years_passed < 70 }	
		create_leader = {
			class = admiral
			species = owner_main_species
			name = random
			skill = 2
			traits = {
				trait = random_trait
			}
		}
	}
    else_if = { limit = { years_passed < 90 }		
		create_leader = {
			class = admiral
			species = owner_main_species
			name = random
			skill = 3
			traits = {
				trait = leader_trait_gale_speed
				trait = random_trait
			}
		}
	}
    else_if = { limit = { years_passed < 110 }		
		create_leader = {
			class = admiral
			species = owner_main_species
			name = random
			skill = 4
			traits = {
				trait = leader_trait_gale_speed
				trait = random_trait
			}
		}
	}
    else = { 	
		create_leader = {
			class = admiral
			species = owner_main_species
			name = random
			skill = 5
			traits = {
				trait = leader_trait_gale_speed
				trait = random_trait
			}
		}
	}	
		create_fleet = {
			name = "NAME_Pirate_Fleet"
			effect = {
				set_owner = event_target:pirate_band
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value < 8
						}
					}
					create_pirate_ships_0 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 8
							trade_intercepted_value < 16
						}
					}
					create_pirate_ships_1 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 16
							trade_intercepted_value < 24
						}
					}
					create_pirate_ships_2 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 24
							trade_intercepted_value < 33
						}
					}
					create_pirate_ships_3 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 33
							trade_intercepted_value < 42
						}
					}
					create_pirate_ships_4 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 42
							trade_intercepted_value < 51
						}
					}
					create_pirate_ships_5 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 51
							trade_intercepted_value < 61
						}
					}
					create_pirate_ships_6 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 61
							trade_intercepted_value < 71
						}
					}
					create_pirate_ships_7 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 71
							trade_intercepted_value < 81
						}
					}
					create_pirate_ships_8 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 81
							trade_intercepted_value < 92
						}
					}
					create_pirate_ships_9 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 92
							trade_intercepted_value < 103
						}
					}
					create_pirate_ships_10 = yes
				}
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 103
							trade_intercepted_value < 114
						}
					}
					create_pirate_ships_11 = yes
				}		
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 114
							trade_intercepted_value < 126
						}
					}
					create_pirate_ships_12 = yes
				}		
				if = {
					limit = { 
						event_target:pirate_system = { 
							trade_intercepted_value >= 126
						}
					}
					create_pirate_ships_13 = yes
				}					
				assign_leader = last_created_leader
				set_location = event_target:pirate_home
				set_fleet_stance = aggressive
				set_aggro_range_measure_from = self
				set_aggro_range = 800
				set_fleet_flag = small_pirates
			}
		}
	}
}

create_thalassocracy_fleet = {
	event_target:thalassocracy_country = {
		create_leader = {
			class = admiral
			species = owner_main_species
			name = random
			skill = 4
			traits = { trait = random_trait }
		}
		create_fleet = {
			name = "NAME_Thalassocracy_fleet"
			effect = {
				set_owner = event_target:thalassocracy_country
				if = {
					limit = { 
						event_target:thalassocracy_target_system = { 
							has_system_trade_value < 100
						}
					}
					create_thalassocracy_ships_1 = yes
				}
				if = {
					limit = { 
						event_target:thalassocracy_target_system = { 
							has_system_trade_value >= 100
							has_system_trade_value < 140
						}
					}
					create_thalassocracy_ships_2 = yes
				}
				if = {
					limit = { 
						event_target:thalassocracy_target_system = { 
							has_system_trade_value >= 140
						}
					}
					create_thalassocracy_ships_3 = yes
				}
				# set_fleet_formation = {  }
				assign_leader = last_created_leader
				set_location = event_target:thalassocracy_spawn_planet
				set_fleet_stance = aggressive
				set_aggro_range_measure_from = self
				set_aggro_range = 400
				save_global_event_target_as = thalassocracy_fleet
                auto_move_to_planet = {
                    target = event_target:thalassocracy_target_planet
                    clear_auto_move_on_arrival = no
                    arrival_effect = thalassocracy_patrol
                }
			}
		}
	
	}

thalassocracy_patrol = {
	if = { #Visit every planet in a system once
		limit = {
			solar_system = {
				any_system_planet = {
					NOT = { has_planet_flag = thalassocracy_visited }
				}
			}
		}
		solar_system = {
			random_system_planet = {
				limit = {
					NOT = { has_planet_flag = thalassocracy_visited }
				}
				save_event_target_as = thalassocracy_target_planet
				set_planet_flag = thalassocracy_visited
			}
		}
	}
	else_if = { #Leave for the next unvisited system
		limit = {
			solar_system = {
				any_neighbor_system = {
					any_system_planet = {
						NOT = { has_planet_flag = thalassocracy_visited }
					}
				}
			}
		}
		solar_system = {
			random_neighbor_system = {
				limit = {
					any_system_planet = {
						NOT = { has_planet_flag = thalassocracy_visited }
					}
				}
				save_global_event_target_as = thalassocracy_target_system
				random_system_planet = {
					limit = {
						NOT = { has_planet_flag = thalassocracy_visited }
					}
					save_event_target_as = thalassocracy_target_planet
					set_planet_flag = thalassocracy_visited
				}
			}
		}
	}
	else = {
		event_target:thalassocracy_spawn_planet = {
			save_event_target_as = thalassocracy_target_planet
		}
	}
	auto_move_to_planet = {
		target = event_target:thalassocracy_target_planet
		clear_auto_move_on_arrival = no
		arrival_effect = thalassocracy_patrol
	}
}
