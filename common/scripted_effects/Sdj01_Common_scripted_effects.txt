            divide_variable = { which = SdjVarPriorityDummy_job_deviant_drone value = 8 }

            #log = "SdjVarJobPriority_job_deviant_drone = [This.GetValue.SdjVarJobPriority_job_deviant_drone] (Pre-Rounded-Up)"
            
            ### Round up
            set_variable = { which = SdjVarPriorityDummy value = 0 }

                limit = { check_variable = { which = SdjVarJobPriority_job_criminal value > 0.01 } }
                change_variable = { which = SdjVarJobPriority_job_criminal value = -1 }
                change_variable = { which = SdjVarPriorityDummy value = 1 }

            set_variable = { which = SdjVarJobPriority_job_criminal value = SdjVarPriorityDummy }
            set_variable = { which = SdjVarPriorityDummy value = 0 }
            while = {
                limit = { check_variable = { which = SdjVarJobPriority_job_deviant_drone value > 0.01 } }
                change_variable = { which = SdjVarJobPriority_job_deviant_drone value = -1 }
                change_variable = { which = SdjVarPriorityDummy value = 1 }
            }
            set_variable = { which = SdjVarJobPriority_job_deviant_drone value = SdjVarPriorityDummy }
            set_variable = { which = SdjVarJobPriority_job_corrupt_drone value = SdjVarPriorityDummy }
            set_variable = { which = SdjVarPriorityDummy value = 0 }
            #log = "SdjVarJobPriority_job_criminal      = [This.GetValue.SdjVarJobPriority_job_criminal] (Final Output)"
            #log = "SdjVarJobPriority_job_deviant_drone = [This.GetValue.SdjVarJobPriority_job_deviant_drone] (Final Output)"
            #log = "SdjVarJobPriority_job_corrupt_drone = [This.GetValue.SdjVarJobPriority_job_corrupt_drone] (Final Output)"
        }
