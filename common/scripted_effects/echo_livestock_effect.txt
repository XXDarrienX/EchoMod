clear_livestock_buildings = {
	if = {
        limit = {
			owner = {
                OR = {  
                has_policy_flag = slavery_allowed
                has_authority = auth_hive_mind
                }
            }
        }  
		every_tile = {
			limit = {
					has_building = "ktlivestock_pen" 				
				OR = {
					has_pop = no
					pop = {
						NAND = {
							is_robot_pop = no
							is_sapient = yes
							is_growing = no
						}
						is_livestock = no
					}					
				}				
			}
			remove_building = yes
		}
	}
}

add_livestock_buildings = {
	if = {
        limit = {
			owner = {
                OR = {  
                has_policy_flag = slavery_allowed
                has_authority = auth_hive_mind
                }
            }
        }  
		every_tile = {
			limit = {	
				NOT = {
					has_building = "ktlivestock_pen" 				
				}
				has_pop = yes
				pop = {
					is_robot_pop = no
					is_sapient = yes
					is_growing = no	
					is_livestock = yes				
				}									
			}
			set_building = "ktlivestock_pen"
		}
	}
}
