autobuild_effect_deliver_pop = {
	if = {
		limit = {
			has_pop = no
		}
		root = {
			random_tile = {
				limit = {
					or = {
						has_building = no
						autobuild_trigger_has_any_regular_building = yes
					}
					autobuild_trigger_tile_has_pop = yes
				}
				resettle_pop = {
					pop = pop
					tile = prevprev
				}
			}
		}
	}
}
